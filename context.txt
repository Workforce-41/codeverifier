        measurementId: "G-B1GQ31753T"
    };

    firebase.initializeApp(firebaseConfig);
    const db = firebase.firestore();

    // Function to fetch and populate movies for each genre
    function fetchAndPopulateMovies(collectionName, carouselId) {
        const carouselInner = document.getElementById(carouselId + '-inner');

        let movieHtml = '';
        let movieCount = 0;

        db.collection(collectionName).get().then((querySnapshot) => {
            querySnapshot.forEach((doc, index) => {
                const movieData = doc.data();

                // Determine the card class based on collectionName
                let cardClass = '';
                if (collectionName === 'action') {
                    cardClass = 'action-card';
                } else if (collectionName === 'drama') {
                    cardClass = 'drama-card';
                } else if (collectionName === 'comedy') {
                    cardClass = 'comedy-card';
                }

                // Start a new carousel-item every 4 movies
                if (movieCount % 4 === 0) {
                    if (movieCount === 0) {
                        movieHtml += `<div class="carousel-item active"><div class="row">`;
                    } else {
                        movieHtml += `</div></div><div class="carousel-item"><div class="row">`;
                    }
                }

                // Append movie HTML with dynamically assigned card class
                movieHtml += `
                    <div class="col-md-3">
                        <div class="card ${cardClass}">
                            <img src="${movieData.image}" class="card-img-top" alt="${movieData.title}">
                            <div class="card-body">
                                <h5 class="card-title">${movieData.title}</h5>
                                <p class="card-text">${movieData.description}</p>
                                <a href="${movieData.link}" class="btn btn-primary" target="_blank">More</a>
                            </div>
                        </div>
                    </div>
                `;
                movieCount++;
            });

            // Close the last row and carousel-item div
            movieHtml += `</div></div>`;

            // Update carousel inner HTML
            carouselInner.innerHTML = movieHtml;

        }).catch((error) => {
            console.error("Error fetching movies: ", error);
        });
    }

    // Fetch and populate movies for each genre
    fetchAndPopulateMovies('action', 'action-carousel');
    fetchAndPopulateMovies('drama', 'drama-carousel');
    fetchAndPopulateMovies('comedy', 'comedy-carousel');
</script>
